1] WAQTD NAMES OF AN EMP HIRED AFTER 1981 AND BEFORE 1987.
		SELECT ENAME 
		FROM EMP 
		WHERE HIREDATE BETWEEN '01-JAN-82' AND '31-DEC-86';
 
2] WAQTD DETAILS OF AN EMP EXCLUDING THE EMP'S WORKING IN DEPTNO 10 OR 20.
		SELECT ENAME
		FROM EMP
		WHERE DEPTNO NOT IN(10,20);
 
3] WAQTD DETILS OF AN EMP EXCEPT PRESIDENT OR MANAGER.
		SELECT *
		FROM EMP
		WHERE JOB NOT IN ('PRESIDENT','MANAGER');
 
4] WAQTD NAMES OF AN EMP WORKING IN DEPTNO 10 OR 20 EXCEPT CLERK OR SALESMAN.
		SELECT ENAME
		FROM EMP
		WHERE DEPTNO IN(10,20) AND 
		JOB NOT IN('CLARK','SALESMAN');
 
5] WAQTD DETAILS OF EMP EXCLUDING THE EMP'S HIRED IN THE YEAR 1981.
		SELECT *
		FROM EMP
		WHERE HIREDATE NOT BETWEEN '01-JAN-81' AND '31-DEC-81';
		
6] WAQTD NAMES OF AN EMP HIRED AFTER 1980 & BEFORE 1987 EXCLUDING THE EMP'S HIRED IN THE YEAR 1982.
		SELECT ENAME
		FROM EMP
		WHERE HIREDATE BETWEEN '01-JAN-81' AND '31-DEC-86' AND
			  HIREDATE NOT BETWEEN '01-JAN-82' AND '31-DEC-82';
 
7] WAQTD NAMES OF AN EMP WORKING AS PRESIDENT OR ANALYST IN DEPTNO 10 OR 20.
		SELECT ENAME
		FROM EMP
		WHERE JOB IN ('PRESIDENT','ANALYST') AND 
		DEPTNO IN(10,20);
 
8] WAQTD DETAILS OF AN EMP HIRED IN THE YEAR 1987.
		SELECT *
		FROM EMP
		WHERE  HIREDATE BETWEEN '01-JAN-87' AND '31-DEC-87';
		
  9] WAQTD NAMES OF AN EMP WORKING AS MANAGER OR ANALYST OR PRESIDENT IN DEPTNO 10 OR 20 & EARNING MORE THAN 2900.
		SELECT ENAME 
		FROM EMP
		WHERE JOB IN ('MANAGER','ANALYST','PRESIDENT') AND
			  DEPTNO IN (10,20) AND
			  SAL > 2900 ;
 
10] WAQTD NAMES OF AN EMP WORKING AS A SALESMAN OR MANAGER IN DEPT NO 10 OR 30 & HIRED IN THE YEAR 1981 & GETS THE COMMISSION OF 1400.
		SELECT ENAME 
		FROM EMP
		WHERE JOB IN ('SALESMAN','MANAGER') AND
		      DEPTNO IN(10,30) AND
			  HIREDATE BETWEEN '01-JAN-81' AND '31-DEC-81' AND
			  COMM =1400;

11] WAQTD NAMES OF AN EMP WORKING AS A PRESIDENT OR SALESMAN OR CLERK 
IN DEPTNO 10 OR 30 & HIRED AFTER 1980 BEFORE 1983 & EARNS SALARY MORE THAN 3000.
		SELECT ENAME 
		FROM EMP
		WHERE JOB IN ('PRESIDENT','SALESMAN','CLARK') AND
			  DEPTNO IN (10,30) AND
			  HIREDATE BETWEEN '01-JAN-81' AND '31-DEC-81' AND
			  SAL > 3000;
 
12] WAQTD DETAILS OF EMP ALONG WITH ANNUAL SAL IF THE 
EMP IS WORKING AS MANAGER OR PRESIDENT IN DEPTNO 10 OR 
30 & HIRED IN THE YEAR 1981 & HE GETS THE SALARY 2850RPS.
		SELECT EMP.*, SAL*12 AS ANNUAL_SAL
		FROM EMP
		WHERE JOB IN ('MANAGER','PRESIDENT') AND
		      DEPTNO IN (10,30) AND
			  HIREDATE BETWEEN '01-JAN-81' AND '31-DEC-81' AND
			  SAL = 2850;
