IN OPERATOR
1] WAQTD NAMES OF AN EMP WHO IS WORKING IN DEPTNO 10,20.
		SELECT ENAME 
		FROM EMP
		WHERE DEPTNO IN (10,20);
		
2] WAQTD SAL AND NAME OF EMP'S WHO IS NOT WORKING IN DEPTNO 10,30.
		SELECT SAL, ENAME
		FROM EMP
		WHERE DEPTNP IN (10,20);
		
3] WAQTD NAME & HIREDATE OF EMP'S WHOSE DESIGNATION IS CLERK,SALESMAN,MANAGER.
		SELECT ENAME, HIREDATE
		FROM EMP 
		WHERE JOB IN ('CLERK','SALESMAN','MANAGER');

4] WAQTD NAME & MANAGER NUMBER OF EMP'S EXCEPT FORD,MILLER,KING,JONES.
		SELECT ENAME, MGR
		FROM EMP
		WHERE ENAME NOT IN ('FORD','MILLER','KING','JONES');
 
5] WAQTD DETAILS OF AN EMP WHO IS EARNING 800,3000,2950.
		SELECT *
		FROM EMP
		WHERE SAL IN(800,3000,2950);
  
6] WAQTD DETAILS OF AN EMP ALONG WITH ANNUAL SALARY IF EMP IS EARNING COMMISSION 300,1400,500.
		SELECT *
		FROM EMP
		WHERE COMM IN(300,1400,500);

7] WAQTD NAME , SAL,ANNUAL SAL IF EMP IS NOT WORKING AS SALESMAN OR MANAGER IN DEPTNO 30.
		SELECT ENAME, SAL, SAL*12 ANNUAL_SAL
		FROM EMP
		WHERE JOB  NOT IN ('SALESMAN','MANAGER') AND 
			  DEPTNO=30;
 
8] WAQTD NAME & COMMISSION OF EMP'S WHO IS HIRED ON 3/12/81 OR 28/9/81.
		SELECT ENAME, COMM 
		FROM EMP
		WHERE HIREDATE IN ('3/DEC/81', '28/SEP/81');
 
9] WAQTD DETAILS OF AN EMP WORKING AS A MANAGER OR ANALYST EXCEPT DEPTNO 30.
		SELECT *
		FROM EMP
		WHERE JOB IN ('MANAGER','ANALYST') AND 
			  DEPTNO !=30;
 
10] WAQTF DETAILS OF AN EMP WHOSE MANAGER NUMBER IS 7902 OR 7566.
		SELECT *
		FROM EMP
		WHERE MGR IN (7902, 7566);
